git init
#+END_SRC

git clone url/filepath repo.git
#+END_SRC

#+END_SRC

* stage,commit,push, tags

#+BEGIN_SRC sh
echo "bin/" > .gitignore
echo "*.o" >> .gitignore
git add README.md
git add -A .
git reset
git commit -m "1st commit"
git remote add origin git@github.com:/username/repo.git

git tag -a v0.1.2 -m “version0.1.2”
git tag -a v0.1.1 9fbc3d0 -m "version0.1.1"

git push origin master
#+END_SRC

* branches

#+BEGIN_SRC sh
git checkout -b feature
git checkout master
git checkout master~2 Makefile
rm -f hello.c
git checkout hello.c
git stash
git stash list
git stash pop
#+END_SRC

* rebase

#+BEGIN_SRC sh
git rebase -i origin/master
git rebase -i HEAD~3
#+END_SRC

=rebase can be used to combine multiple commits into single one before push to remote repo=
